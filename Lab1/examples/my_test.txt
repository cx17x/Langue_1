method bubble_sort()
var arr: array[] of int;
    n, i, j, tmp: int;
    swapped: bool;
begin
    arr := int(8);
    arr[0] := 43;
    arr[1] := 12;
    arr[2] := 99;
    arr[3] := 5;
    arr[4] := 70;
    arr[5] := 18;
    arr[6] := 33;
    arr[7] := 1;
    n := 8;

    i := 0;
    while i < n do begin
        swapped := false;
        j := 0;
        while j < n - i - 1 do begin
            if arr[j] > arr[j + 1] then begin
                tmp := arr[j];
                arr[j] := arr[j + 1];
                arr[j + 1] := tmp;
                swapped := true;
            end;
            j := j + 1;
        end;
        if swapped = false then
            break;
        i := i + 1;
    end;

    i := 0;
    while i < n do begin
        send_byte(int_to_byte(arr[i]));
        i := i + 1;
    end;
end;

method main()
begin
    bubble_sort();
end;
